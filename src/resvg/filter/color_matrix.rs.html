5use super::{f32_bound, ImageRefMut};
6use rgb::RGBA8;
7use usvg::filter::ColorMatrixKind as ColorMatrix;
8
9/// Applies a color matrix filter.
10///
11/// Input image pixels should have an **unpremultiplied alpha**.
12pub fn apply(matrix: &ColorMatrix, src: ImageRefMut) {
13    match matrix {
14        ColorMatrix::Matrix(m) => {
15            for pixel in src.data {
16                let (r, g, b, a) = to_normalized_components(*pixel);
17
18                let new_r = r * m[0] + g * m[1] + b * m[2] + a * m[3] + m[4];
19                let new_g = r * m[5] + g * m[6] + b * m[7] + a * m[8] + m[9];
20                let new_b = r * m[10] + g * m[11] + b * m[12] + a * m[13] + m[14];
21                let new_a = r * m[15] + g * m[16] + b * m[17] + a * m[18] + m[19];
22
23                pixel.r = from_normalized(new_r);
24                pixel.g = from_normalized(new_g);
25                pixel.b = from_normalized(new_b);
26                pixel.a = from_normalized(new_a);
27            }
28        }
29        ColorMatrix::Saturate(v) => {
30            let v = v.get().max(0.0);
31            let m = [
32                0.213 + 0.787 * v,
33                0.715 - 0.715 * v,
34                0.072 - 0.072 * v,
35                0.213 - 0.213 * v,
36                0.715 + 0.285 * v,
37                0.072 - 0.072 * v,
38                0.213 - 0.213 * v,
39                0.715 - 0.715 * v,
40                0.072 + 0.928 * v,
41            ];
42
43            for pixel in src.data {
44                let (r, g, b, _) = to_normalized_components(*pixel);
45
46                let new_r = r * m[0] + g * m[1] + b * m[2];
47                let new_g = r * m[3] + g * m[4] + b * m[5];
48                let new_b = r * m[6] + g * m[7] + b * m[8];
49
50                pixel.r = from_normalized(new_r);
51                pixel.g = from_normalized(new_g);
52                pixel.b = from_normalized(new_b);
53            }
54        }
55        ColorMatrix::HueRotate(angle) => {
56            let angle = angle.to_radians();
57            let a1 = angle.cos();
58            let a2 = angle.sin();
59            let m = [
60                0.213 + 0.787 * a1 - 0.213 * a2,
61                0.715 - 0.715 * a1 - 0.715 * a2,
62                0.072 - 0.072 * a1 + 0.928 * a2,
63                0.213 - 0.213 * a1 + 0.143 * a2,
64                0.715 + 0.285 * a1 + 0.140 * a2,
65                0.072 - 0.072 * a1 - 0.283 * a2,
66                0.213 - 0.213 * a1 - 0.787 * a2,
67                0.715 - 0.715 * a1 + 0.715 * a2,
68                0.072 + 0.928 * a1 + 0.072 * a2,
69            ];
70
71            for pixel in src.data {
72                let (r, g, b, _) = to_normalized_components(*pixel);
73
74                let new_r = r * m[0] + g * m[1] + b * m[2];
75                let new_g = r * m[3] + g * m[4] + b * m[5];
76                let new_b = r * m[6] + g * m[7] + b * m[8];
77
78                pixel.r = from_normalized(new_r);
79                pixel.g = from_normalized(new_g);
80                pixel.b = from_normalized(new_b);
81            }
82        }
83        ColorMatrix::LuminanceToAlpha => {
84            for pixel in src.data {
85                let (r, g, b, _) = to_normalized_components(*pixel);
86
87                let new_a = r * 0.2125 + g * 0.7154 + b * 0.0721;
88
89                pixel.r = 0;
90                pixel.g = 0;
91                pixel.b = 0;
92                pixel.a = from_normalized(new_a);
93            }
94        }
95    }
96}
97
98#[inline]
99fn to_normalized_components(pixel: RGBA8) -> (f32, f32, f32, f32) {
100    (
101        pixel.r as f32 / 255.0,
102        pixel.g as f32 / 255.0,
103        pixel.b as f32 / 255.0,
104        pixel.a as f32 / 255.0,
105    )
106}
107
108#[inline]
109fn from_normalized(c: f32) -> u8 {
110    (f32_bound(0.0, c, 1.0) * 255.0) as u8
111}
