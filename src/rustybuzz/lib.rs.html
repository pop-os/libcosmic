27bitflags::bitflags! {
28    /// Flags for buffers.
29    #[derive(Default, Debug, Clone, Copy)]
30    pub struct BufferFlags: u32 {
31        /// Indicates that special handling of the beginning of text paragraph can be applied to this buffer. Should usually be set, unless you are passing to the buffer only part of the text without the full context.
32        const BEGINNING_OF_TEXT             = 1 << 1;
33        /// Indicates that special handling of the end of text paragraph can be applied to this buffer, similar to [`BufferFlags::BEGINNING_OF_TEXT`].
34        const END_OF_TEXT                   = 1 << 2;
35        /// Indicates that characters with `Default_Ignorable` Unicode property should use the corresponding glyph from the font, instead of hiding them (done by replacing them with the space glyph and zeroing the advance width.) This flag takes precedence over [`BufferFlags::REMOVE_DEFAULT_IGNORABLES`].
36        const PRESERVE_DEFAULT_IGNORABLES   = 1 << 3;
37        /// Indicates that characters with `Default_Ignorable` Unicode property should be removed from glyph string instead of hiding them (done by replacing them with the space glyph and zeroing the advance width.) [`BufferFlags::PRESERVE_DEFAULT_IGNORABLES`] takes precedence over this flag.
38        const REMOVE_DEFAULT_IGNORABLES     = 1 << 4;
39        /// Indicates that a dotted circle should not be inserted in the rendering of incorrect character sequences (such as `<0905 093E>`).
40        const DO_NOT_INSERT_DOTTED_CIRCLE   = 1 << 5;
41        /// Indicates that the shape() call and its variants should perform various verification processes on the results of the shaping operation on the buffer. If the verification fails, then either a buffer message is sent, if a message handler is installed on the buffer, or a message is written to standard error. In either case, the shaping result might be modified to show the failed output.
42        const VERIFY                        = 1 << 6;
43        /// Indicates that the `UNSAFE_TO_CONCAT` glyph-flag should be produced by the shaper. By default it will not be produced since it incurs a cost.
44        const PRODUCE_UNSAFE_TO_CONCAT      = 1 << 7;
45    }
46}

68        /// Do not serialize glyph cluster.
69        const NO_CLUSTERS       = 0b00000001;
70        /// Do not serialize glyph position information.
71        const NO_POSITIONS      = 0b00000010;
72        /// Do no serialize glyph name.
73        const NO_GLYPH_NAMES    = 0b00000100;
74        /// Serialize glyph extents.
75        const GLYPH_EXTENTS     = 0b00001000;
76        /// Serialize glyph flags.
77        const GLYPH_FLAGS       = 0b00010000;
78        /// Do not serialize glyph advances, glyph offsets will reflect absolute
79        /// glyph positions.
80        const NO_ADVANCES       = 0b00100000;
