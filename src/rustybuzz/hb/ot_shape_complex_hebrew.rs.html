51fn compose(ctx: &hb_ot_shape_normalize_context_t, a: char, b: char) -> Option<char> {
52    // Hebrew presentation-form shaping.
53    // https://bugzilla.mozilla.org/show_bug.cgi?id=728866
54    // Hebrew presentation forms with dagesh, for characters U+05D0..05EA;
55    // Note that some letters do not have a dagesh presForm encoded.
56    match unicode::compose(a, b) {
57        Some(c) => Some(c),
58        None if !ctx.plan.has_gpos_mark => {
59            // Special-case Hebrew presentation forms that are excluded from
60            // standard normalization, but wanted for old fonts.
61            let a = a as u32;
62            let b = b as u32;
63            match b {
64                0x05B4 => {
65                    // HIRIQ
66                    match a {
67                        0x05D9 => Some('\u{FB1D}'), // YOD
68                        _ => None,
69                    }
70                }
71                0x05B7 => {
72                    // patah
73                    match a {
74                        0x05D9 => Some('\u{FB1F}'), // YIDDISH YOD YOD
75                        0x05D0 => Some('\u{FB2E}'), // ALEF
76                        _ => None,
77                    }
78                }
79                0x05B8 => {
80                    // QAMATS
81                    match a {
82                        0x05D0 => Some('\u{FB2F}'), // ALEF
83                        _ => None,
84                    }
85                }
86                0x05B9 => {
87                    // HOLAM
88                    match a {
89                        0x05D5 => Some('\u{FB4B}'), // VAV
90                        _ => None,
91                    }
92                }
93                0x05BC => {
94                    // DAGESH
95                    match a {
96                        0x05D0..=0x05EA => {
97                            let c = S_DAGESH_FORMS[a as usize - 0x05D0];
98                            if c != '\0' {
99                                Some(c)
100                            } else {
101                                None
102                            }
103                        }
104                        0xFB2A => Some('\u{FB2C}'), // SHIN WITH SHIN DOT
105                        0xFB2B => Some('\u{FB2D}'), // SHIN WITH SIN DOT
106                        _ => None,
107                    }
108                }
109                0x05BF => {
110                    // RAFE
111                    match a {
112                        0x05D1 => Some('\u{FB4C}'), // BET
113                        0x05DB => Some('\u{FB4D}'), // KAF
114                        0x05E4 => Some('\u{FB4E}'), // PE
115                        _ => None,
116                    }
117                }
118                0x05C1 => {
119                    // SHIN DOT
120                    match a {
121                        0x05E9 => Some('\u{FB2A}'), // SHIN
122                        0xFB49 => Some('\u{FB2C}'), // SHIN WITH DAGESH
123                        _ => None,
124                    }
125                }
126                0x05C2 => {
127                    // SIN DOT
128                    match a {
129                        0x05E9 => Some('\u{FB2B}'), // SHIN
130                        0xFB49 => Some('\u{FB2D}'), // SHIN WITH DAGESH
131                        _ => None,
132                    }
133                }
134                _ => None,
135            }
136        }
137        None => None,
138    }
139}
