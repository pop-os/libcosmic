4//! Contains stylesheet implementation for [`crate::widget::segmented_button`].
5
6use crate::widget::segmented_button::{Appearance, ItemAppearance, StyleSheet};
7use crate::{theme::Theme, widget::segmented_button::ItemStatusAppearance};
8use iced::Border;
9use iced_core::{Background, border::Radius};
10use palette::WithAlpha;
11
12#[derive(Default)]
13pub enum SegmentedButton {
14    /// A tabbed widget for switching between views in an interface.
15    #[default]
16    TabBar,
17    /// A widget for multiple choice selection.
18    Control,
19    /// Navigation bar style
20    NavBar,
21    /// Or implement any custom theme of your liking.
22    Custom(Box<dyn Fn(&Theme) -> Appearance>),
23}
24
25impl StyleSheet for Theme {
26    type Style = SegmentedButton;
27
28    #[allow(clippy::too_many_lines)]
29    fn horizontal(&self, style: &Self::Style) -> Appearance {
30        let cosmic = self.cosmic();
31        let container = self.current_container();
32        match style {
33            SegmentedButton::Control => {
34                let rad_xl = cosmic.corner_radii.radius_xl;
35                let rad_0 = cosmic.corner_radii.radius_0;
36                let active = horizontal::selection_active(cosmic, &container.component);
37                Appearance {
38                    background: Some(Background::Color(container.component.base.into())),
39                    border: Border {
40                        radius: rad_xl.into(),
41                        ..Default::default()
42                    },
43                    inactive: ItemStatusAppearance {
44                        background: None,
45                        first: ItemAppearance {
46                            border: Border {
47                                radius: Radius::from([rad_xl[0], rad_0[1], rad_0[2], rad_xl[3]]),
48                                ..Default::default()
49                            },
50                        },
51                        middle: ItemAppearance {
52                            border: Border {
53                                radius: cosmic.corner_radii.radius_0.into(),
54                                ..Default::default()
55                            },
56                        },
57                        last: ItemAppearance {
58                            border: Border {
59                                radius: Radius::from([rad_0[0], rad_xl[1], rad_xl[2], rad_0[3]]),
60                                ..Default::default()
61                            },
62                        },
63                        text_color: container.component.on.into(),
64                    },
65                    hover: hover(cosmic, &active, 0.2),
66                    pressed: hover(cosmic, &active, 0.15),
67                    active,
68                    ..Default::default()
69                }
70            }
71
72            SegmentedButton::NavBar => Appearance {
73                active_width: 0.0,
74                ..horizontal::tab_bar(cosmic, container)
75            },
76
77            SegmentedButton::TabBar => horizontal::tab_bar(cosmic, container),
78
79            SegmentedButton::Custom(func) => func(self),
80        }
81    }
82

139mod horizontal {
140    use super::Appearance;
141    use crate::widget::segmented_button::{ItemAppearance, ItemStatusAppearance};
142    use cosmic_theme::{Component, Container};
143    use iced::Border;
144    use iced_core::{Background, border::Radius};
145    use palette::WithAlpha;
146
147    pub fn tab_bar(cosmic: &cosmic_theme::Theme, container: &Container) -> Appearance {
148        let active = tab_bar_active(cosmic);
149        let hc = cosmic.is_high_contrast;
150        let border = if hc {
151            Border {
152                color: container.component.border.into(),
153                radius: cosmic.corner_radii.radius_0.into(),
154                width: 1.0,
155            }
156        } else {
157            Border::default()
158        };
159
160        Appearance {
161            active_width: 4.0,
162            border: Border {
163                radius: cosmic.corner_radii.radius_0.into(),
164                ..Default::default()
165            },
166            inactive: ItemStatusAppearance {
167                background: None,
168                first: ItemAppearance { border },
169                middle: ItemAppearance { border },
170                last: ItemAppearance { border },
171                text_color: container.component.on.into(),
172            },
173            hover: super::hover(cosmic, &active, 0.3),
174            pressed: super::hover(cosmic, &active, 0.25),
175            active,
176            ..Default::default()
177        }
178    }
179
180    pub fn selection_active(
181        cosmic: &cosmic_theme::Theme,
182        component: &Component,
183    ) -> ItemStatusAppearance {
184        let rad_xl = cosmic.corner_radii.radius_xl;
185        let rad_0 = cosmic.corner_radii.radius_0;
186
187        ItemStatusAppearance {
188            background: Some(Background::Color(
189                cosmic.palette.neutral_5.with_alpha(0.1).into(),
190            )),
191            first: ItemAppearance {
192                border: Border {
193                    radius: Radius::from([rad_xl[0], rad_0[1], rad_0[2], rad_xl[3]]),
194                    ..Default::default()
195                },
196            },
197            middle: ItemAppearance {
198                border: Border {
199                    radius: cosmic.corner_radii.radius_0.into(),
200                    ..Default::default()
201                },
202            },
203            last: ItemAppearance {
204                border: Border {
205                    radius: Radius::from([rad_0[0], rad_xl[1], rad_xl[2], rad_0[3]]),
206                    ..Default::default()
207                },
208            },
209            text_color: cosmic.accent_text_color().into(),
210        }
211    }
