3use std::time::Duration;

5const LAG: f32 = 0.1;
6
7#[derive(Default)]
8pub struct Progress {
9    pub current: f32,
10    target: Option<f32>,
11    last: Option<Instant>,
12}
13
14impl Progress {
15    /// Smoothly chases `target` using exponential decay.
16    /// Returns `true` if a redraw should be requested.
17    pub fn update(&mut self, target: f32, now: Instant) -> bool {
18        // Don't animate on start
19        let Some(last) = self.last else {
20            self.current = target;
21            self.target = Some(target);
22            self.last = Some(now);
23            return false;
24        };
25
26        // Sync animation clock when target changes
27        if self.target != Some(target) {
28            self.target = Some(target);
29            self.last = Some(now);
30            return true;
31        }
32
33        let dt = (now - last).as_secs_f32();
34        self.last = Some(now);
35        let diff = target - self.current;
36
37        if diff.abs() > 0.001 {
38            self.current += diff * (1.0 - (-dt / LAG).exp());
39            true
40        } else {
41            self.current = target;
42            false
43        }
44    }
45}

47#[derive(Clone, Copy)]
48pub struct Animation {
49    expanding: bool,
50    start: Instant,
51    last: Instant,
52    offset: u32,
53}
54
55impl Default for Animation {
56    fn default() -> Self {
57        let now = Instant::now();
58        Self {
59            expanding: true,
60            start: now,
61            last: now,
62            offset: 0,
63        }
64    }
65}
66
67impl Animation {
68    pub fn timed_transition(
69        &self,
70        cycle_duration: Duration,
71        period: Duration,
72        wrap: f32,
73        now: Instant,
74    ) -> Self {
75        let additional =
76            ((now - self.last).as_secs_f32() / period.as_secs_f32() * u32::MAX as f32) as u32;
77        let new_offset = self.offset.wrapping_add(additional);
78
79        if !cycle_duration.is_zero() && now.duration_since(self.start) > cycle_duration {
80            let offset = if self.expanding {
81                new_offset
82            } else {
83                new_offset.wrapping_add((wrap * u32::MAX as f32) as u32)
84            };
85            Self {
86                expanding: !self.expanding,
87                start: now,
88                last: now,
89                offset,
90            }
91        } else {
92            Self {
93                last: now,
94                offset: new_offset,
95                ..*self
96            }
97        }
98    }
99
100    pub fn bar_positions(&self, cycle_duration: Duration, min: f32, wrap: f32) -> (f32, f32) {
101        let offset = self.offset as f32 / u32::MAX as f32;
102        let progress = if !cycle_duration.is_zero() {
103            smootherstep(
104                self.last.duration_since(self.start).as_secs_f32() / cycle_duration.as_secs_f32(),
105            )
106        } else {
107            1.0
108        };
109        if self.expanding {
110            (offset, offset + min + wrap * progress)
111        } else {
112            (offset + wrap * progress, offset + min + wrap)
113        }
114    }
