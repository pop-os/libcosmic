9/// A collect of moment integrals for a path, used for computing
10/// shape properties like area, centroid, and inertia.
11///
12/// The moments are computed using Green's theorem, which relates the
13/// integrals of a function over a closed curve to the integrals of its
14/// partial derivatives over the region enclosed by the curve.
15#[derive(Debug, Default, Copy, Clone)]
16pub struct Moments {
17    /// The first moment of area about the x-axis.
18    pub moment_x: f64,
19    /// The first moment of area about the y-axis.
20    pub moment_y: f64,
21    /// The second moment of area about the x-axis (also known as the
22    /// area moment of inertia about the x-axis).
23    pub moment_xx: f64,
24    /// The second moment of area about the xy-axis (also known as the
25    /// product of inertia).
26    pub moment_xy: f64,
27    /// The second moment of area about the y-axis (also known as the
28    /// area moment of inertia about the y-axis).
29    pub moment_yy: f64,
30}
31
32impl core::ops::Add<Self> for Moments {
33    type Output = Self;
34
35    fn add(self, rhs: Self) -> Self::Output {
36        Self {
37            moment_x: self.moment_x + rhs.moment_x,
38            moment_y: self.moment_y + rhs.moment_y,
39            moment_xx: self.moment_xx + rhs.moment_xx,
40            moment_xy: self.moment_xy + rhs.moment_xy,
41            moment_yy: self.moment_yy + rhs.moment_yy,
42        }
43    }
44}

49    fn handle_line(&mut self, p0: Point, p1: Point) {
50        let (x0, y0) = (p0.x, p0.y);
51        let (x1, y1) = (p1.x, p1.y);
52        let r0 = x1 * y0;
53        let r1 = x1 * y1;
54        let r2 = x1.powi(2);
55        let r3 = r2 * y1;
56        let r4 = y0 - y1;
57        let r5 = r4 * x0;
58        let r6 = x0.powi(2);
59        let r7 = 2.0 * y0;
60        let r8 = y0.powi(2);
61        let r9 = y1.powi(2);
62        let r10 = x1.powi(3);
63        let r11 = y0.powi(3);
64        let r12 = y1.powi(3);
65        self.moment_x += -r2 * y0 / 6.0 - r3 / 3.0 - r5 * x1 / 6.0 + r6 * (r7 + y1) / 6.0;
66        self.moment_y +=
67            -r0 * y1 / 6.0 - r8 * x1 / 6.0 - r9 * x1 / 6.0 + x0 * (r8 + r9 + y0 * y1) / 6.0;
68        self.moment_xx += -r10 * y0 / 12.0 - r10 * y1 / 4.0 - r2 * r5 / 12.0 - r4 * r6 * x1 / 12.0
69            + x0.powi(3) * (3.0 * y0 + y1) / 12.0;
70        self.moment_xy += -r2 * r8 / 24.0 - r2 * r9 / 8.0 - r3 * r7 / 24.0
71            + r6 * (r7 * y1 + 3.0 * r8 + r9) / 24.0
72            - x0 * x1 * (r8 - r9) / 12.0;
73        self.moment_yy += -r0 * r9 / 12.0 - r1 * r8 / 12.0 - r11 * x1 / 12.0 - r12 * x1 / 12.0
74            + x0 * (r11 + r12 + r8 * y1 + r9 * y0) / 12.0;
75    }

522/// A trait for types that can provide moment integrals for a path using Green's theorem.
523pub trait ParamCurveMoments<'a> {
524    /// Returns the moment integrals for the path using Green's theorem.
525    fn moments(&'a self) -> Moments;
526}
527
528impl<'a, T: 'a> ParamCurveMoments<'a> for T
529where
530    &'a T: IntoIterator<Item = PathEl>,
531{
532    fn moments(&'a self) -> Moments {
533        let mut moments = Moments::default();
534        let mut start_pt: Point = Point::ZERO;
535        let mut cur: Point = Point::ZERO;
536        for el in self {
537            match el {
538                PathEl::MoveTo(p) => {
539                    start_pt = p;
540                    cur = p;
541                }
542                PathEl::LineTo(p) => {
543                    moments.handle_line(cur, p);
544                    cur = p;
545                }
546                PathEl::QuadTo(p0, p1) => {
547                    moments.handle_quad(cur, p0, p1);
548                    cur = p1;
549                }
550                PathEl::CurveTo(p1, p2, p3) => {
551                    moments.handle_cubic(cur, p1, p2, p3);
552                    cur = p3;
553                }
554                PathEl::ClosePath => {
555                    if cur != start_pt {
556                        moments.handle_line(cur, start_pt);
557                        cur = start_pt;
558                    }
559                }
560            }
561        }
562        moments
563    }
564}
