142    /// write the CSS to the appropriate directory
143    /// Should be written in the XDG config directory for gtk-4.0
144    ///
145    /// # Errors
146    ///
147    /// Returns an `OutputError` if there is an error writing the CSS file.
148    #[cold]
149    pub fn write_gtk4(&self) -> Result<(), OutputError> {
150        let css_str = self.as_gtk4();
151        let Some(mut config_dir) = dirs::config_dir() else {
152            return Err(OutputError::MissingConfigDir);
153        };
154
155        let name = if self.is_dark {
156            "dark.css"
157        } else {
158            "light.css"
159        };
160
161        config_dir.extend(["gtk-4.0", "cosmic"]);
162        if !config_dir.exists() {
163            std::fs::create_dir_all(&config_dir).map_err(OutputError::Io)?;
164        }
165
166        let mut file = File::create(config_dir.join(name)).map_err(OutputError::Io)?;
167        file.write_all(css_str.as_bytes())
168            .map_err(OutputError::Io)?;
169
170        Ok(())
171    }

173    /// Apply gtk color variable settings
174    ///
175    /// # Errors
176    ///
177    /// Returns an `OutputError` if there is an error applying the CSS file.
178    #[cold]
179    pub fn apply_gtk(is_dark: bool) -> Result<(), OutputError> {
180        let Some(config_dir) = dirs::config_dir() else {
181            return Err(OutputError::MissingConfigDir);
182        };
183
184        let mut gtk4 = config_dir.join("gtk-4.0");
185        let mut gtk3 = config_dir.join("gtk-3.0");
186
187        fs::create_dir_all(&gtk4).map_err(OutputError::Io)?;
188        fs::create_dir_all(&gtk3).map_err(OutputError::Io)?;
189
190        let cosmic_css_dir = gtk4.join("cosmic");
191        let cosmic_css = cosmic_css_dir.join(if is_dark { "dark.css" } else { "light.css" });
192
193        gtk4.push("gtk.css");
194        gtk3.push("gtk.css");
195
196        #[cfg(target_family = "unix")]
197        for gtk_dest in [&gtk4, &gtk3] {
198            use std::os::unix::fs::symlink;
199            Self::backup_non_cosmic_css(gtk_dest, &cosmic_css_dir).map_err(OutputError::Io)?;
200
201            if gtk_dest.exists() {
202                fs::remove_file(gtk_dest).map_err(OutputError::Io)?;
203            }
204
205            symlink(&cosmic_css, gtk_dest).map_err(OutputError::Io)?;
206        }
207        Ok(())
208    }

223        let gtk4_dest = gtk4.join("gtk.css");
224        let cosmic_css = gtk4.join("cosmic");

227        let res = Self::reset_cosmic_css(&gtk3_dest, &cosmic_css).map_err(OutputError::Io);
228        Self::reset_cosmic_css(&gtk4_dest, &cosmic_css).map_err(OutputError::Io)?;
229        res
230    }

232    #[cold]
233    fn backup_non_cosmic_css(path: &Path, cosmic_css: &Path) -> io::Result<()> {
234        if !Self::is_cosmic_css(path, cosmic_css)?.unwrap_or(true) {
235            let backup_path = path.with_extension("css.bak");
236            fs::rename(path, &backup_path)?;
237        }
238        Ok(())
239    }
240
241    #[cold]
242    fn reset_cosmic_css(path: &Path, cosmic_css: &Path) -> io::Result<()> {
243        if Self::is_cosmic_css(path, cosmic_css)?.unwrap_or_default() {
244            fs::remove_file(path)?;
245        }
246        Ok(())
247    }
248
249    fn is_cosmic_css(path: &Path, cosmic_css: &Path) -> io::Result<Option<bool>> {
250        if !path.exists() {
251            return Ok(None);
252        }
253
254        if let Ok(metadata) = fs::symlink_metadata(path) {
255            if metadata.file_type().is_symlink() {
256                if let Ok(actual_cosmic_css) = fs::read_link(path) {
257                    let canonical_target = fs::canonicalize(&actual_cosmic_css)?;
258                    let canonical_base = fs::canonicalize(cosmic_css)?;
259                    return Ok(Some(
260                        canonical_target == canonical_base
261                            || canonical_target.starts_with(&canonical_base),
262                    ));
263                }
264            }
265        }
266        Ok(Some(false))
267    }
