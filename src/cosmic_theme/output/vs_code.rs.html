268impl Theme {
269    #[cold]
270    pub fn apply_vs_code(self) -> Result<(), OutputError> {
271        let vs_theme = VsTheme::from(self);
272        let mut config_dir = dirs::config_dir().ok_or(OutputError::MissingConfigDir)?;
273        config_dir.extend(["Code", "User"]);
274        let vs_code_dir = config_dir;
275        if !vs_code_dir.exists() {
276            std::fs::create_dir_all(&vs_code_dir).map_err(OutputError::Io)?;
277        }
278
279        // just add the json entry for workbench.colorCustomizations
280        let settings_file = vs_code_dir.join("settings.json");
281        let settings = std::fs::read_to_string(&settings_file).unwrap_or_default();
282        let mut settings: serde_json::Value = serde_json::from_str(&settings)?;
283        settings["workbench.colorCustomizations"] = serde_json::to_value(vs_theme).unwrap();
284        settings["window.autoDetectColorScheme"] = serde_json::Value::Bool(true);
285        std::fs::write(
286            &settings_file,
287            serde_json::to_string_pretty(&settings).unwrap(),
288        )
289        .map_err(OutputError::Io)?;
290
291        Ok(())
292    }
293
294    #[cold]
295    pub fn reset_vs_code() -> Result<(), OutputError> {
296        let mut config_dir = dirs::config_dir().ok_or(OutputError::MissingConfigDir)?;
297        config_dir.extend(["Code", "User", "settings.json"]);
298        let settings_file = config_dir;
299        // just remove the json entry for workbench.colorCustomizations
300        let settings = std::fs::read_to_string(&settings_file).unwrap_or_default();
301        let mut settings: serde_json::Value = serde_json::from_str(&settings).unwrap_or_default();
302        settings["workbench.colorCustomizations"] = serde_json::Value::Null;
303
304        std::fs::write(
305            &settings_file,
306            serde_json::to_string_pretty(&settings).unwrap(),
307        )
308        .map_err(OutputError::Io)?;
309
310        Ok(())
311    }
