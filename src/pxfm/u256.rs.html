36// /// A 256-bit unsigned integer represented as two 128-bit native-endian limbs.
37// #[allow(non_camel_case_types)]
38// #[derive(Clone, Copy, Debug)]
39// pub(crate) struct u256 {
40//     pub lo: u128,
41//     pub hi: u128,
42// }
43//
44// impl PartialEq for u256 {
45//     #[inline]
46//     fn eq(&self, other: &Self) -> bool {
47//         self.hi == other.hi && self.lo == other.lo
48//     }
49// }

51// impl Eq for u256 {}

53// impl PartialOrd for u256 {
54//     #[inline]
55//     fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
56//         Some(self.cmp(other))
57//     }

60// impl Ord for u256 {
61//     #[inline]
62//     fn cmp(&self, other: &Self) -> Ordering {
63//         match self.hi.cmp(&other.hi) {
64//             Ordering::Equal => self.lo.cmp(&other.lo),
65//             ord => ord,
66//         }
67//     }

70// pub(crate) fn mulhi_u256(a: u256, b: u256) -> u256 {
71//     // // Products
72//     let lo_lo = a.lo.expanding_mul(b.lo); // a.lo * b.lo
73//     let lo_hi = a.lo.expanding_mul(b.hi); // a.lo * b.hi
74//     let hi_lo = a.hi.expanding_mul(b.lo); // a.hi * b.lo
75//     let hi_hi = a.hi.expanding_mul(b.hi); // a.hi * b.hi
76//
77//     let carry = u256::from_u128(lo_lo.hi)
78//         .add(u256::from_u128(lo_hi.lo))
79//         .add(u256::from_u128(hi_lo.lo));
80//     let mid = u256::from_u128(lo_hi.hi)
81//         .add(u256::from_u128(hi_lo.hi))
82//         .add(u256::from_u128(carry.hi));
83//     hi_hi.add(mid)
84// }

89//         // [(self.hi >> 64) as u64, (self.hi & 0xffff_ffff_ffff_ffff) as u64, (self.lo >> 64) as u64, (self.lo & 0xffff_ffff_ffff_ffff) as u64]
90//         let mut out = [0u8; 32];
91//         out[16..32].copy_from_slice(&self.hi.to_le_bytes());
92//         out[0..16].copy_from_slice(&self.lo.to_le_bytes());
93//         out
94//     }

151// macro_rules! impl_common {
152//     ($ty:ty) => {
153//         impl ops::BitOr for $ty {
154//             type Output = Self;
155//
156//             fn bitor(mut self, rhs: Self) -> Self::Output {
157//                 self.lo |= rhs.lo;
158//                 self.hi |= rhs.hi;
159//                 self
160//             }
161//         }
162//
163//         impl ops::Not for $ty {
164//             type Output = Self;
165//
166//             fn not(mut self) -> Self::Output {
167//                 self.lo = !self.lo;
168//                 self.hi = !self.hi;
169//                 self
170//             }
171//         }
172//
173//         impl ops::Shl<u32> for $ty {
174//             type Output = Self;
175//
176//             fn shl(self, _rhs: u32) -> Self::Output {
177//                 unimplemented!("only used to meet trait bounds")
178//             }
179//         }
180//     };

183// impl u256 {
184//     #[inline]
185//     pub(crate) const fn wrapping_shl(self, shift: u32) -> Self {
186//         match shift {
187//             0 => self,
188//             s if s < 128 => {
189//                 let hi = (self.hi << s) | (self.lo >> (128 - s));
190//                 let lo = self.lo << s;
191//                 u256 { lo, hi }
192//             }
193//             s if s < 256 => {
194//                 let lo = 0;
195//                 let hi = self.lo << (s - 128);
196//                 u256 { lo, hi }
197//             }
198//             _ => u256 { lo: 0, hi: 0 },
199//         }
200//     }
201//
202//     #[inline]
203//     pub(crate) fn wrapping_sub(self, rhs: Self) -> Self {
204//         let (lo, carry) = self.lo.overflowing_sub(rhs.lo);
205//         let hi = self.hi.wrapping_sub(carry as u128).wrapping_sub(rhs.hi);
206//
207//         Self { lo, hi }
208//     }
209//
210//     #[inline]
211//     pub(crate) fn overflowing_add(self, rhs: Self) -> (Self, bool) {
212//         let (lo, carry_lo) = self.lo.overflowing_add(rhs.lo);
213//         let (hi_intermediate, carry_hi1) = self.hi.overflowing_add(rhs.hi);
214//         let (hi, carry_hi2) = hi_intermediate.overflowing_add(carry_lo as u128);
215//
216//         let overflow = carry_hi1 || carry_hi2;
217//         (Self { lo, hi }, overflow)
218//     }
219// }

243// impl ops::Shr<u32> for u256 {
244//     type Output = Self;
245//
246//     #[inline]
247//     fn shr(mut self, rhs: u32) -> Self::Output {
248//         debug_assert!(rhs < 256, "attempted to shift right with overflow");
249//         if rhs >= 256 {
250//             return Self { lo: 0, hi: 0 };
251//         }
252//
253//         if rhs == 0 {
254//             return self;
255//         }
256//
257//         if rhs < 128 {
258//             self.lo >>= rhs;
259//             self.lo |= self.hi << (128 - rhs);
260//         } else {
261//             self.lo = self.hi >> (rhs - 128);
262//         }
263//
264//         if rhs < 128 {
265//             self.hi >>= rhs;
266//         } else {
267//             self.hi = 0;
268//         }
269//
270//         self
271//     }
272// }

286// trait WideningMul {
287//     type Output;
288//     fn expanding_mul(self, rhs: Self) -> Self::Output;
289// }
290//
291// impl WideningMul for u128 {
292//     type Output = u256;
293//
294//     #[inline]
295//     fn expanding_mul(self, rhs: Self) -> u256 {
296//         let l0 = self & U128_LO_MASK;
297//         let l1 = rhs & U128_LO_MASK;
298//         let h0 = self >> 64;
299//         let h1 = rhs >> 64;
300//
301//         let p_ll: u128 = l0.wrapping_mul(l1);
302//         let p_lh: u128 = l0.wrapping_mul(h1);
303//         let p_hl: u128 = h0.wrapping_mul(l1);
304//         let p_hh: u128 = h0.wrapping_mul(h1);
305//
306//         let s0 = p_hl + (p_ll >> 64);
307//         let s1 = (p_ll & U128_LO_MASK) + (s0 << 64);
308//         let s2 = p_lh + (s1 >> 64);
309//
310//         let lo = (p_ll & U128_LO_MASK) + (s2 << 64);
311//         let hi = p_hh + (s0 >> 64) + (s2 >> 64);
312//
313//         u256 { lo, hi }
314//     }
315// }
