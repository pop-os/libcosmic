29use crate::common::f_fmla;
30use crate::logs::{fast_logf, simple_fast_log};
31use crate::polyeval::{f_polyeval3, f_polyeval5, f_polyeval7, f_polyeval8};
32use crate::sin_cosf::fast_sinpif;
33
34/// Computes log(gamma(x))
35///
36/// ulp 0.5
37pub fn f_lgammaf(x: f32) -> f32 {
38    if (x.to_bits() & 0x0007_ffff) == 0 {
39        if x.is_infinite() {
40            return f32::INFINITY;
41        }
42        if x.is_nan() {
43            return f32::NAN;
44        }
45        if x == 0. {
46            return 1. / x;
47        }
48    }
49
50    if x.floor() == x {
51        if x == 2. || x == 1. {
52            return 0.;
53        }
54        if x.is_sign_negative() {
55            return f32::INFINITY;
56        }
57    }
58
59    let ax = f32::from_bits(x.to_bits() & 0x7fff_ffff);
60    let dx = ax as f64;
61
62    let is_positive = x.is_sign_positive();
63    let mut sum_parity = 1f64;
64
65    let mut f_res = 0f64;
66
67    // For negative x, since (G is gamma function)
68    // -x*G(-x)*G(x) = pi/sin(pi*x),
69    // we have
70    // G(x) = pi/(sin(pi*x)*(-x)*G(-x))
71    // since G(-x) is positive, sign(G(x)) = sign(sin(pi*x)) for x<0
72    // Hence, for x<0, signgam = sign(sin(pi*x)) and
73    // lgamma(x) = log(|Gamma(x)|)
74    //  = log(pi/(|x*sin(pi*x)|)) - lgamma(-x);
75    if !is_positive {
76        let y1 = ax.floor();
77        let fraction = ax - y1; // excess over the boundary
78
79        let a = fast_sinpif(fraction);
80
81        sum_parity = -1.;
82        const LOG_PI: f64 = f64::from_bits(0x3ff250d048e7a1bd);
83        f_res = LOG_PI - simple_fast_log(a * dx);
84    }
85
86    if ax <= 0.007 {
87        // Log[Gamma(x)] = -x.log() - euler * x + (zeta(2)/2)*x^2 - (zeta(3)/3)*x^3
88        let fp = f_polyeval3(
89            dx,
90            f64::from_bits(0xbfe2788cfc6fb619),
91            f64::from_bits(0x3fea51a6625307d3),
92            f64::from_bits(0xbfd9a4d55beab2d7),
93        );
94        let d_log = simple_fast_log(dx);
95        let log_gamma = f_fmla(fp, dx, -d_log);
96        f_res = f_fmla(log_gamma, sum_parity, f_res);
97    } else if ax < 1. {
98        // Poly for loggamma(x + 1) - log(x) generated by Wolfram Mathematica
99        // <<FunctionApproximations`
100        // ClearAll["Global`*"]
101        // f[x_]:=LogGamma[x+1]
102        // {err0, approx}=MiniMaxApproximation[f[z],{z,{0.06,0.9999999},6,6},WorkingPrecision->90]
103        // num=Numerator[approx][[1]];
104        // den=Denominator[approx][[1]];
105        // poly=den;
106        // coeffs=CoefficientList[poly,z];
107        // TableForm[Table[Row[{"'",NumberForm[coeffs[[i+1]],{50,50}, ExponentFunction->(Null&)],"',"}],{i,0,Length[coeffs]-1}]]
108        let num = f_polyeval7(
109            dx,
110            f64::from_bits(0xbd24e4cf78c8818a),
111            f64::from_bits(0xbfe2788cfc6f64e6),
112            f64::from_bits(0xbfe1ea632fe853b2),
113            f64::from_bits(0x3fd988d2daad3806),
114            f64::from_bits(0x3fe1f4870eaafdf4),
115            f64::from_bits(0x3fc51e12d5b97330),
116            f64::from_bits(0x3f889ebeb9f0c8ff),
117        );
118        let den = f_polyeval7(
119            dx,
120            f64::from_bits(0x3ff0000000000000),
121            f64::from_bits(0x4003289872066f33),
122            f64::from_bits(0x4000373d88c32fe5),
123            f64::from_bits(0x3fe71e95bc874164),
124            f64::from_bits(0x3fb9936a0e008be7),
125            f64::from_bits(0x3f6cba20a1988a60),
126            f64::from_bits(0xbef3ca884ba4237a),
127        );
128        f_res = f_fmla(num / den - fast_logf(ax), sum_parity, f_res);
129    } else if ax <= 4. {
130        // Poly for loggamma(x + 1) - log(x) generated by Wolfram Mathematica
131        // <<FunctionApproximations`
132        // ClearAll["Global`*"]
133        // f[x_]:=LogGamma[x+1]
134        // {err0, approx}=MiniMaxApproximation[f[z],{z,{1.0000001,4},7,7},WorkingPrecision->90]
135        // num=Numerator[approx][[1]];
136        // den=Denominator[approx][[1]];
137        // poly=den;
138        // coeffs=CoefficientList[poly,z];
139        // TableForm[Table[Row[{"'",NumberForm[coeffs[[i+1]],{50,50}, ExponentFunction->(Null&)],"',"}],{i,0,Length[coeffs]-1}]]
140        let num = f_polyeval8(
141            dx,
142            f64::from_bits(0xbe630847302a205f),
143            f64::from_bits(0xbfe2788c24a7deff),
144            f64::from_bits(0xbfdd0f61b8171907),
145            f64::from_bits(0x3fdab4ec27801e42),
146            f64::from_bits(0x3fde11fd49427a3c),
147            f64::from_bits(0x3fc0dbce39c660da),
148            f64::from_bits(0x3f88e0b4cd3e97f7),
149            f64::from_bits(0x3f328fcd9e9ee94e),
150        );
151        let den = f_polyeval8(
152            dx,
153            f64::from_bits(0x3ff0000000000000),
154            f64::from_bits(0x4001b135a31bffae),
155            f64::from_bits(0x3ffbbecb5e39f4ea),
156            f64::from_bits(0x3fe2e4da095d10bd),
157            f64::from_bits(0x3fb63cb7548d0d30),
158            f64::from_bits(0x3f73a67613163399),
159            f64::from_bits(0x3f10edd4e2b80e6f),
160            f64::from_bits(0xbe77f4e4068f9d32),
161        );
162        f_res = f_fmla(num / den - fast_logf(ax), sum_parity, f_res);
163    } else if ax <= 12. {
164        // Poly for loggamma(x) generated by Wolfram Mathematica
165        // <<FunctionApproximations`
166        // ClearAll["Global`*"]
167        // f[x_]:=LogGamma[x]
168        // {err0, approx}=MiniMaxApproximation[f[z],{z,{4,12},7,7},WorkingPrecision->90]
169        // num=Numerator[approx][[1]];
170        // den=Denominator[approx][[1]];
171        // poly=den;
172        // coeffs=CoefficientList[poly,z];
173        // TableForm[Table[Row[{"'",NumberForm[coeffs[[i+1]],{50,50}, ExponentFunction->(Null&)],"',"}],{i,0,Length[coeffs]-1}]]
174        let num = f_polyeval8(
175            dx,
176            f64::from_bits(0x40086069fcc21690),
177            f64::from_bits(0x4008c8ef8602e78d),
178            f64::from_bits(0xc0186323149757ae),
179            f64::from_bits(0xbff6a9ded42d3f31),
180            f64::from_bits(0x3ff1adb5566cd807),
181            f64::from_bits(0x3fcff023390dead6),
182            f64::from_bits(0x3f8977a5454ec393),
183            f64::from_bits(0x3f21361088f694dc),
184        );
185        let den = f_polyeval8(
186            dx,
187            f64::from_bits(0x3ff0000000000000),
188            f64::from_bits(0x4016096b397dbe73),
189            f64::from_bits(0x401478926593a395),
190            f64::from_bits(0x3ff6a931f9e31511),
191            f64::from_bits(0x3fc0fbe57bf263a4),
192            f64::from_bits(0x3f7023f51ba39a98),
193            f64::from_bits(0x3efabd75ebfbde40),
194            f64::from_bits(0xbe4b2c9e2e4e7daa),
195        );
196        f_res = f_fmla(num / den, sum_parity, f_res);
197    } else {
198        // Stirling's approximation of Log(Gamma) and then Exp[Log[Gamma]]
199        let y_recip = 1. / dx;
200        let y_sqr = y_recip * y_recip;
201        // Bernoulli coefficients generated by SageMath:
202        // var('x')
203        // def bernoulli_terms(x, N):
204        //     S = 0
205        //     for k in range(1, N+1):
206        //         B = bernoulli(2*k)
207        //         term = (B / (2*k*(2*k-1))) * x**((2*k-1))
208        //         S += term
209        //     return S
210        //
211        // terms = bernoulli_terms(x, 5)
212        let bernoulli_poly = f_polyeval5(
213            y_sqr,
214            f64::from_bits(0x3fb5555555555555),
215            f64::from_bits(0xbf66c16c16c16c17),
216            f64::from_bits(0x3f4a01a01a01a01a),
217            f64::from_bits(0xbf43813813813814),
218            f64::from_bits(0x3f4b951e2b18ff23),
219        );
220        // Log[Gamma(x)] = x*log(x) - x + 1/2*Log(2*PI/x) + bernoulli_terms
221        const LOG2_PI_OVER_2: f64 = f64::from_bits(0x3fed67f1c864beb5);
222        let mut log_gamma = f_fmla(bernoulli_poly, y_recip, -dx) + LOG2_PI_OVER_2;
223        log_gamma = f_fmla(fast_logf(ax), dx - 0.5, log_gamma);
224        f_res = f_fmla(log_gamma, sum_parity, f_res);
225    }
226
227    f_res as f32
228}

230#[cfg(test)]
231mod tests {
232    use super::*;
233
234    #[test]
235    fn test_lgammaf() {
236        assert_eq!(
237            f_lgammaf(0.000000000000000000000000000000000000000015425),
238            93.97255
239        );
240        assert_eq!(f_lgammaf(1.7506484), -0.0842405);
241        assert_eq!(f_lgammaf(7.095007e-8), 16.461288);
242        assert!(f_lgammaf(-12.).is_infinite());
243        assert_eq!(f_lgammaf(2.), 0.);
244        assert_eq!(f_lgammaf(1.), 0.);
245        assert_eq!(f_lgammaf(0.53), 0.5156078);
246        assert_eq!(f_lgammaf(1.53), -0.11927056);
247        assert_eq!(f_lgammaf(4.53), 2.4955146);
248        assert_eq!(f_lgammaf(11.77), 16.94281);
249        assert_eq!(f_lgammaf(22.77), 47.756233);
250        assert_eq!(f_lgammaf(-0.53), 1.2684484);
251        assert_eq!(f_lgammaf(-1.53), 0.84318066);
252        assert_eq!(f_lgammaf(-4.53), -2.8570588);
253        assert_eq!(f_lgammaf(-11.77), -17.850103);
254        assert_eq!(f_lgammaf(-22.77), -49.323418);
255        assert_eq!(f_lgammaf(f32::NEG_INFINITY), f32::INFINITY);
256    }
