67#[cold]
68fn exp1m1f_accurate(ux: u32, z: f64, sv: u64, ia: f64) -> f32 {
69    if ux > 0xc18aa123u32 {
70        // x < -17.32
71        return -1.0 + f32::from_bits(0x32800000);
72    }
73    const ILN2H: f64 = f64::from_bits(0x4047154765000000);
74    const ILN2L: f64 = f64::from_bits(0x3e55c17f0bbbe880);
75    let s = f64::from_bits(sv);
76    let h = f_fmla(ILN2H, z, -ia) + ILN2L * z;
77    let h2 = h * h;
78    let w = s * h;
79    const CH: [u64; 6] = [
80        0x3f962e42fefa39ef,
81        0x3f2ebfbdff82c58f,
82        0x3ebc6b08d702e0ed,
83        0x3e43b2ab6fb92e5e,
84        0x3dc5d886e6d54203,
85        0x3d4430976b8ce6ef,
86    ];
87
88    let h0 = f_fmla(h, f64::from_bits(CH[5]), f64::from_bits(CH[4]));
89    let h1 = f_fmla(h, f64::from_bits(CH[3]), f64::from_bits(CH[2]));
90    let h2t = f_fmla(h, f64::from_bits(CH[1]), f64::from_bits(CH[0]));
91
92    let t0 = f_fmla(h2, h0, h1);
93    let t1 = f_fmla(h2, t0, h2t);
94
95    let r = f_fmla(w, t1, s - 1.);
96    r as f32
97}

99#[cold]
100fn expm1f_small(z: f64) -> f32 {
101    const B: [u64; 8] = [
102        0x3fdfffffffffffc2,
103        0x3fc55555555555fe,
104        0x3fa555555559767f,
105        0x3f81111111098dc1,
106        0x3f56c16bca988aa9,
107        0x3f2a01a07658483f,
108        0x3efa05b04d2c3503,
109        0x3ec71de3a960b5e3,
110    ];
111    let z2 = z * z;
112    let z4 = z2 * z2;
113
114    let r0 = f_fmla(z, f64::from_bits(B[7]), f64::from_bits(B[6]));
115    let r1 = f_fmla(z, f64::from_bits(B[5]), f64::from_bits(B[4]));
116    let r2 = f_fmla(z, f64::from_bits(B[3]), f64::from_bits(B[2]));
117    let r3 = f_fmla(z, f64::from_bits(B[1]), f64::from_bits(B[0]));
118
119    let w0 = f_fmla(z2, r0, r1);
120    let w1 = f_fmla(z2, r2, r3);
121
122    let q0 = f_fmla(z4, w0, w1);
123
124    let r = f_fmla(z2, q0, z);
125    r as f32
126}

128/// Computes e^x - 1
129///
130/// Max ULP 0.5
131#[inline]
132pub fn f_expm1f(x: f32) -> f32 {
133    const ILN2: f64 = f64::from_bits(0x40471547652b82fe);
134    const BIG: f64 = f64::from_bits(0x4338000000000000);
135    let t = x.to_bits();
136    let z = x as f64;
137    let ux = t;
138    let ax = ux.wrapping_shl(1);
139    if ax < 0x7c400000u32 {
140        // |x| < 0.15625
141        if ax < 0x676a09e8u32 {
142            // |x| < 0x1.6a09e8p-24
143            if ax == 0x0u32 {
144                return x;
145            } // x = +-0
146            let res = dd_fmlaf(x.abs(), f32::from_bits(0x33000000), x);
147            return res;
148        }
149        return expm1f_small(z);
150    }
151    if ax >= 0x8562e430u32 {
152        // |x| > 88.72
153        if ax > (0xffu32 << 24) {
154            return x + x;
155        } // nan
156        if ux >> 31 != 0 {
157            // x < 0
158            if ax == (0xffu32 << 24) {
159                return -1.0;
160            }
161            return black_box(-1.0) + black_box(f32::from_bits(0x32800000));
162        }
163        if ax == (0xffu32 << 24) {
164            return f32::INFINITY;
165        }
166        let r = f64::from_bits(0x47efffffe0000000) * z;
167        return r as f32;
168    }
169    let a = ILN2 * z;
170    let ia = a.round_ties_even();
171    let h = a - ia;
172    let h2 = h * h;
173    let u = (ia + BIG).to_bits();
174    const C: [u64; 4] = [
175        0x3ff0000000000000,
176        0x3f962e42fef4c4e7,
177        0x3f2ebfd1b232f475,
178        0x3ebc6b19384ecd93,
179    ];
180
181    let c2 = f_fmla(h, f64::from_bits(C[3]), f64::from_bits(C[2]));
182    let c0 = f_fmla(h, f64::from_bits(C[1]), f64::from_bits(C[0]));
183    let tdl = TD[(u & 0x1f) as usize];
184    let sv: u64 = tdl.wrapping_add((u >> 5).wrapping_shl(52));
185    let r = f_fmla(h2, c2, c0) * f64::from_bits(sv) - 1.0;
186    let ub: f32 = r as f32;
187    let lb = (r - f64::from_bits(sv) * f64::from_bits(0x3de3b30000000000)) as f32;
188
189    // Ziv's accuracy test
190    if ub != lb {
191        return exp1m1f_accurate(ux, z, sv, ia);
192    }
193    ub
194}

197mod tests {
198    use crate::f_expm1f;

200    #[test]
201    fn test_expm1f() {
202        assert_eq!(f_expm1f(2.213121), 8.144211);
203        assert_eq!(f_expm1f(-3.213121), -0.9597691);
204        assert_eq!(f_expm1f(-2.35099e-38), -2.35099e-38);
205        assert_eq!(
206            f_expm1f(0.00000000000000000000000000000000000004355616),
207            0.00000000000000000000000000000000000004355616
208        );
