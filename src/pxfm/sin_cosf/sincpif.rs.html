29use crate::common::f_fmla;
30use crate::polyeval::f_polyeval5;
31use crate::sin_cosf::sincosf_eval::sincospif_eval;
32
33/// Computes sin(PI*x)/(PI*x)
34///
35/// Produces normalized sinc.
36///
37/// ulp 0.5
38pub fn f_sincpif(x: f32) -> f32 {
39    let x_abs = x.to_bits() & 0x7fff_ffffu32;
40    let xd = x as f64;
41
42    // |x| <= 1/16
43    if x_abs <= 0x3d80_0000u32 {
44        // |x| < 0.0000009546391
45        if x_abs < 0x3580_2126u32 {
46            if x_abs == 0u32 {
47                return 1.;
48            }
49            const M_SQR_PI_OVER_6: f64 = f64::from_bits(0xbffa51a6625307d3);
50
51            // Small values approximated with Taylor poly
52            // sincpi(x) ~ 1 - x^2*Pi^2/6 + O(x^4)
53            let x2 = xd * xd;
54            let p = f_fmla(x2, M_SQR_PI_OVER_6, 1.);
55            return p as f32;
56        }
57
58        let xsqr = xd * xd;
59
60        // Generated by Sollya:
61        // d = [0, 1/16];
62        // f_sincpif = sin(y*pi)/(y*pi);
63        // Q = fpminimax(f_sincpif, [|0, 2, 4, 6, 8|], [|D...|], d, relative, floating);
64        // See ./notes/sincpif.sollya
65        let p = f_polyeval5(
66            xsqr,
67            f64::from_bits(0x3ff0000000000000),
68            f64::from_bits(0xbffa51a662530723),
69            f64::from_bits(0x3fe9f9cb401e8e85),
70            f64::from_bits(0xbfc86a8da89c9234),
71            f64::from_bits(0x3f9ac0a16798157e),
72        );
73        return p as f32;
74    }
75
76    // Numbers greater or equal to 2^23 are always integers or NaN
77    // integers are always 0
78    if x_abs >= 0x4b00_0000u32 || x.floor() == x {
79        if x_abs >= 0x7f80_0000u32 {
80            return x + f32::NAN;
81        }
82        return if x.is_sign_negative() { -0. } else { 0. };
83    }
84
85    const PI: f64 = f64::from_bits(0x400921fb54442d18);
86    let rs = sincospif_eval(xd);
87    let sf = f_fmla(rs.sin_y, rs.cos_k, f_fmla(rs.cosm1_y, rs.sin_k, rs.sin_k));
88    (sf / (PI * xd)) as f32
89}

91#[cfg(test)]
92mod tests {
93    use super::*;
94
95    #[test]
96    fn test_sincpif_eval() {
97        assert_eq!(f_sincpif(1.0), 0.);
98        assert_eq!(f_sincpif(2.0), 0.);
99        assert_eq!(f_sincpif(3.0), 0.);
100        assert_eq!(f_sincpif(0.0543242), 0.99515265);
101        assert_eq!(f_sincpif(0.002134242), 0.9999925);
102        assert_eq!(f_sincpif(0.00000005421321), 1.0);
103        assert!(f_sincpif(f32::INFINITY).is_nan());
104        assert!(f_sincpif(f32::NEG_INFINITY).is_nan());
105        assert!(f_sincpif(f32::NAN).is_nan());
106    }
107}
