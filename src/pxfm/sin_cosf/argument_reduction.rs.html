32#[derive(Debug, Copy, Clone)]
33pub(crate) struct ArgumentReducer {
34    pub(crate) x: f64,
35    pub(crate) x_abs: u32,
36}

38impl ArgumentReducer {
39    // Return k and y, where
40    // k = round(x * 32 / pi) and y = (x * 32 / pi) - k.
41    #[cfg(any(
42        all(
43            any(target_arch = "x86", target_arch = "x86_64"),
44            target_feature = "fma"
45        ),
46        all(target_arch = "aarch64", target_feature = "neon")
47    ))]
48    #[inline]
49    pub(crate) fn reduce_small(self) -> (f64, i64) {
50        /*
51           Generated in SageMath:
52           z = RealField(300)(32) / RealField(300).pi()
53           n = 53
54           x_hi = RealField(n)(z)  # convert to f64
55           x_mid = RealField(n)(z - RealField(300)(x_hi))
56           x_lo = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid))
57           print(double_to_hex(x_hi), ",")
58           print(double_to_hex(x_mid), ",")
59           print(double_to_hex(x_lo), ",")
60        */
61        const THIRTYTWO_OVER_PI: [u64; 3] =
62            [0x40245f306dc9c883, 0xbcc6b01ec5417056, 0xb966447e493ad4ce];
63
64        let kd = (self.x * f64::from_bits(THIRTYTWO_OVER_PI[0])).round();
65        let mut y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[0]), -kd);
66        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[1]), y);
67        (y, kd as i64)
68    }
69
70    // Return k and y, where
71    // k = round(x * 32 / pi) and y = (x * 32 / pi) - k.
72    #[cfg(not(any(
73        all(
74            any(target_arch = "x86", target_arch = "x86_64"),
75            target_feature = "fma"
76        ),
77        all(target_arch = "aarch64", target_feature = "neon")
78    )))]
79    #[inline]
80    pub(crate) fn reduce_small(self) -> (f64, i64) {
81        /*
82           Generated in SageMath:
83           z = RealField(300)(32) / RealField(300).pi()
84           n = 28
85           x_hi = RealField(n)(z)  # convert to f64
86           x_mid = RealField(n)(z - RealField(300)(x_hi))
87           x_lo = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid))
88           print(double_to_hex(x_hi), ",")
89           print(double_to_hex(x_mid), ",")
90           print(double_to_hex(x_lo), ",")
91        */
92        const THIRTYTWO_OVER_PI: [u64; 3] =
93            [0x40245f306e000000, 0xbe3b1bbeae000000, 0x3c63f84eb0000000];
94        let prod = self.x * f64::from_bits(THIRTYTWO_OVER_PI[0]);
95        let kd = prod.round();
96        let mut y = prod - kd;
97        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[1]), y);
98        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[2]), y);
99        (y, kd as i64)
100    }
101
102    // Return k and y, where
103    // k = round(x * 32 / pi) and y = (x * 32 / pi) - k.
104    #[cfg(any(
105        all(
106            any(target_arch = "x86", target_arch = "x86_64"),
107            target_feature = "fma"
108        ),
109        all(target_arch = "aarch64", target_feature = "neon")
110    ))]
111    #[inline]
112    pub(crate) fn reduce_large(self, exponent: i32) -> (f64, i64) {
113        /*
114        Generated in SageMath:
115        z = RealField(300)(32) / RealField(300).pi()
116        n = 53
117        x_hi = RealField(n)(z)  # convert to f64
118        x_mid = RealField(n)(z - RealField(300)(x_hi))
119        x_lo = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid))
120        x_lo_2 = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo))
121        x_lo_3 = z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo) - RealField(300)(x_lo_2)
122        print(double_to_hex(x_hi), ",")
123        print(double_to_hex(x_mid), ",")
124        print(double_to_hex(x_lo), ",")
125        print(double_to_hex(x_lo_2), ",")
126        print(double_to_hex(x_lo_3), ",")
127         */
128        const THIRTYTWO_OVER_PI: [u64; 5] = [
129            0x40245f306dc9c883,
130            0xbcc6b01ec5417056,
131            0xb966447e493ad4ce,
132            0x360e21c820ff28b2,
133            0xb29508510ea79237,
134        ];
135
136        // 2^45 <= |x| < 2^99
137        if exponent < 99 {
138            // - When x < 2^99, the full exact product of x * THIRTYTWO_OVER_PI[0]
139            // contains at least one integral bit <= 2^5.
140            // - When 2^45 <= |x| < 2^55, the lowest 6 unit bits are contained
141            // in the last 12 bits of double(x * THIRTYTWO_OVER_PI[0]).
142            // - When |x| >= 2^55, the LSB of double(x * THIRTYTWO_OVER_PI[0]) is at
143            // least 2^6.
144            let mut prod_hi = self.x * f64::from_bits(THIRTYTWO_OVER_PI[0]);
145            prod_hi = f64::from_bits(
146                prod_hi.to_bits()
147                    & (if exponent < 55 {
148                        0xfffffffffffff000
149                    } else {
150                        0xffffffffffffffff
151                    }),
152            ); // |x| < 2^55
153            let k_hi = prod_hi.round();
154            let truncated_prod = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[0]), -k_hi);
155            let prod_lo = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[1]), truncated_prod);
156            let k_lo = prod_lo.round();
157            let mut y = f_fmla(
158                self.x,
159                f64::from_bits(THIRTYTWO_OVER_PI[1]),
160                truncated_prod - k_lo,
161            );
162            y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[2]), y);
163            y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[3]), y);
164
165            return (y, k_lo as i64);
166        }
167
168        // - When x >= 2^110, the full exact product of x * THIRTYTWO_OVER_PI[0] does
169        // not contain any of the lowest 6 unit bits, so we can ignore it completely.
170        // - When 2^99 <= |x| < 2^110, the lowest 6 unit bits are contained
171        // in the last 12 bits of double(x * THIRTYTWO_OVER_PI[1]).
172        // - When |x| >= 2^110, the LSB of double(x * THIRTYTWO_OVER_PI[1]) is at
173        // least 64.
174        let mut prod_hi = self.x * f64::from_bits(THIRTYTWO_OVER_PI[1]);
175        prod_hi = f64::from_bits(
176            prod_hi.to_bits()
177                & (if exponent < 110 {
178                    0xfffffffffffff000
179                } else {
180                    0xffffffffffffffff
181                }),
182        ); // |x| < 2^110
183        let k_hi = prod_hi.round();
184        let truncated_prod = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[1]), -k_hi);
185        let prod_lo = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[2]), truncated_prod);
186        let k_lo = prod_lo.round();
187        let mut y = f_fmla(
188            self.x,
189            f64::from_bits(THIRTYTWO_OVER_PI[2]),
190            truncated_prod - k_lo,
191        );
192        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[3]), y);
193        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[4]), y);
194
195        (y, k_lo as i64)
196    }
197
198    // Return k and y, where
199    // k = round(x * 32 / pi) and y = (x * 32 / pi) - k.
200    #[cfg(not(any(
201        all(
202            any(target_arch = "x86", target_arch = "x86_64"),
203            target_feature = "fma"
204        ),
205        all(target_arch = "aarch64", target_feature = "neon")
206    )))]
207    #[inline]
208    pub(crate) fn reduce_large(self, exponent: i32) -> (f64, i64) {
209        // For large range, there are at most 2 parts of THIRTYTWO_OVER_PI_28
210        // contributing to the lowest 6 binary digits (k & 63).  If the least
211        // significant bit of x * the least significant bit of THIRTYTWO_OVER_PI_28[i]
212        // >= 64, we can completely ignore THIRTYTWO_OVER_PI_28[i].
213
214        // Generated by SageMath:
215        // z = RealField(300)(32) / RealField(300).pi()
216        // n = 28
217        // x_hi = RealField(n)(z)  # convert to f64
218        // x_mid = RealField(n)(z - RealField(300)(x_hi))
219        // x_lo = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid))
220        // x_lo_2 = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo))
221        // x_lo_3 = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo) - RealField(300)(x_lo_2))
222        // x_lo_4 = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo) - RealField(300)(x_lo_2) - RealField(300)(x_lo_3))
223        // x_lo_5 = RealField(n)(z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo) - RealField(300)(x_lo_2) - RealField(300)(x_lo_3) - RealField(300)(x_lo_4))
224        // x_lo_6 = (z - RealField(300)(x_hi) - RealField(300)(x_mid) - RealField(300)(x_lo) - RealField(300)(x_lo_2) - RealField(300)(x_lo_3) - RealField(300)(x_lo_4) - RealField(300)(x_lo_5))
225        //
226        //
227        // print(double_to_hex(x_hi), ",")
228        // print(double_to_hex(x_mid), ",")
229        // print(double_to_hex(x_lo), ",")
230        // print(double_to_hex(x_lo_2), ",")
231        // print(double_to_hex(x_lo_3), ",")
232        // print(double_to_hex(x_lo_4), ",")
233        // print(double_to_hex(x_lo_5), ",")
234        // print(double_to_hex(x_lo_6), ",")
235        static THIRTYTWO_OVER_PI: [u64; 8] = [
236            0x40245f306e000000,
237            0xbe3b1bbeae000000,
238            0x3c63f84eb0000000,
239            0xba87056592000000,
240            0x38bc0db62a000000,
241            0xb6e4cd8778000000,
242            0xb51bef806c000000,
243            0x33363abdebbc561b,
244        ];
245
246        let mut idx = 0;
247        const FRACTION_LEN: i32 = 23;
248        let x_lsb_exp_m4 = exponent - FRACTION_LEN;
249
250        // Exponents of the least significant bits of the corresponding entries in
251        // THIRTYTWO_OVER_PI_28.
252        static THIRTYTWO_OVER_PI_28_LSB_EXP: [i32; 8] =
253            [-24, -55, -81, -114, -143, -170, -200, -230];
254
255        // Skipping the first parts of 32/pi such that:
256        //   LSB of x * LSB of THIRTYTWO_OVER_PI_28[i] >= 32.
257        while x_lsb_exp_m4 + THIRTYTWO_OVER_PI_28_LSB_EXP[idx] > 5 {
258            idx += 1;
259        }
260
261        let prod_hi = self.x * f64::from_bits(THIRTYTWO_OVER_PI[idx]);
262        // Get the integral part of x * THIRTYTWO_OVER_PI_28[idx]
263        let k_hi = prod_hi.round();
264        // Get the fractional part of x * THIRTYTWO_OVER_PI_28[idx]
265        let frac = prod_hi - k_hi;
266        let prod_lo = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[idx + 1]), frac);
267        let k_lo = prod_lo.round();
268
269        // Now y is the fractional parts.
270        let mut y = prod_lo - k_lo;
271        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[idx + 2]), y);
272        y = f_fmla(self.x, f64::from_bits(THIRTYTWO_OVER_PI[idx + 3]), y);
273
274        (y, (k_hi as i64).wrapping_add(k_lo as i64))
275    }

277    #[inline]
278    pub(crate) fn reduce(self) -> (f64, i64) {
279        #[cfg(any(
280            all(
281                any(target_arch = "x86", target_arch = "x86_64"),
282                target_feature = "fma"
283            ),
284            all(target_arch = "aarch64", target_feature = "neon")
285        ))]
286        const SMALL_PASS_BOUND: u32 = 0x5600_0000u32;
287        #[cfg(not(any(
288            all(
289                any(target_arch = "x86", target_arch = "x86_64"),
290                target_feature = "fma"
291            ),
292            all(target_arch = "aarch64", target_feature = "neon")
293        )))]
294        const SMALL_PASS_BOUND: u32 = 0x4a80_0000u32;
295        if self.x_abs < SMALL_PASS_BOUND {
296            // 2^45
297            self.reduce_small()
298        } else {
299            self.reduce_large(get_exponent_f32(f32::from_bits(self.x_abs)))
300        }
301    }
