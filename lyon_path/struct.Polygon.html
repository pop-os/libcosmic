Expand description

A view over a sequence of endpoints forming a polygon.

ยงExample

use lyon_path::polygon::Polygon;
use lyon_path::geom::point;

let path = Polygon {
    points: &[
        point(0.0, 0.0),
        point(10.0, 10.0),
        point(0.0, 10.0),
    ],
    closed: true,
};

for event in path.path_events() {
    // same as iterating a regular `Path` object.
}
