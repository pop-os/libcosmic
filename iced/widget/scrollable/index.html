iced::widget

Module scrollable

Source
Expand description

Scrollables let users navigate an endless amount of content with a scrollbar.

§Example

use iced::widget::{column, scrollable, vertical_space};

enum Message {
    // ...
}

fn view(state: &State) -> Element<'_, Message> {
    scrollable(column![
        "Scroll me!",
        vertical_space().height(3000),
        "You did it!",
    ]).into()
}
